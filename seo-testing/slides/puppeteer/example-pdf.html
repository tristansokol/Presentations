automated PDF-ing 

				const puppeteer = require('puppeteer');

				browser = await puppeteer.launch({
					headless: true,
					slowMo: 80,
				});
				page = await browser.newPage();

				await page.goto('https://google.com/');

				page.emulateMedia('screen')
				await page.pdf({ path:'./report.pdf', printBackground:true })

				browser.close();
